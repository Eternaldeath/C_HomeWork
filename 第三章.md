```c
{
```

第四题
```c
# include <stdio.h>

int main()
{
	char c1,c2;
	c1 = 97;
	c2 = 98;
	printf("c1=%c,c2=%c\n",c1,c2);
	printf("c1=%d,c2=%d\n",c1,c2);
	
	return 0;
 } 

 // 输出
c1=a,c2=b
c1=97,c2=98

原因是，当对 c1，c2 用字符输出是，输出的是 ASCll 码表中该数字对应的字符

// 将 c1 = 197，c2 = 198
# include <stdio.h>

int main()
{
	char c1,c2;
	c1 = 197;
	c2 = 198;
	printf("c1=%c,c2=%c\n",c1,c2);
	printf("c1=%d,c2=%d\n",c1,c2);
	
	return 0;
 } 

 // 输出
c1=�,c2=�
c1=-59,c2=-58

原因是，常规的 ACSll 码表只到 127 位，超出就显示乱码


// 将 char c1，c2; 改为 int c1, c2;
# include <stdio.h>

int main()
{
	int c1,c2;
	c1 = 97;
	c2 = 98;
	printf("c1=%c,c2=%c\n",c1,c2);
	printf("c1=%d,c2=%d\n",c1,c2);
	
	return 0;
 } 

 // 输出
c1=a,c2=b
c1=97,c2=98
```
